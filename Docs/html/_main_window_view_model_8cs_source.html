241  public async Task ChangeEnvironmentAsync(IProgress<CmdExecutorProgress> progress)
242  {
243  SetIsChangeInProgress(true);
244  var cmdExecutor = new CmdExecutor(SelectedEnvironment);
245  cmdExecutor.EnvironmentChanged += OnEnvironmentChanged;
246  var currentEnvironmentName = System.Environment.GetEnvironmentVariable("_CHAMI_ENV");
247  var cancellationToken = GetNewCancellationToken();
248  if (currentEnvironmentName != null)
249  {
250  var currentOsEnvironment = _dataAdapter.GetEnvironmentEntityByName(currentEnvironmentName);
251  // currentOsEnvironment could be null in case there's a stray _CHAMI_ENV environment variable but no
252  // corresponding entity
253  if (currentOsEnvironment != null)
254  {
255  foreach (var environmentVariable in currentOsEnvironment.EnvironmentVariables)
256  {
257  var newCommand =
258  EnvironmentVariableCommandFactory.GetCommand(typeof(EnvironmentVariableRemovalCommand),
259  environmentVariable);
260  cmdExecutor.AddCommand(newCommand);
261  }
262  }
263  }
264 
265  var newEnvironment = _dataAdapter.GetEnvironmentEntityById(SelectedEnvironment.Id);
266  cmdExecutor.AddCommand(EnvironmentVariableCommandFactory.GetCommand(
267  typeof(EnvironmentVariableApplicationCommand),
268  new EnvironmentVariable() {Name = "_CHAMI_ENV", Value = SelectedEnvironment.Name}));
269 
270  foreach (var environmentVariable in newEnvironment.EnvironmentVariables)
271  {
272  var newCommand = EnvironmentVariableCommandFactory.GetCommand(
273  typeof(EnvironmentVariableApplicationCommand),
274  environmentVariable);
275  cmdExecutor.AddCommand(newCommand);
276  }
277 
278  await cmdExecutor.ExecuteAsync(progress, cancellationToken);
279  SetIsChangeInProgress(false);

58  var pInfo = viewModel.GetType().GetProperty(setting.PropertyName);

63  }
64 
65  var settingPInfo = pInfo.PropertyType.GetProperty(setting.SettingName);
66  if (settingPInfo == null)

69  }
70

73  try

84  if (objectWrapper != null)
85  {
86  propertyValue = objectWrapper.Unwrap();
87  }
88  }
89  catch (MissingMethodException)

142  private object GetBoxedConvertedObject(string typeName, string value)
143  {
144  object propertyValue = null;
145  switch (typeName)
146  {
147  case "System.Boolean":
148  case "bool":
149  propertyValue = value.StringToBool();
150  break;
151  case "System.String":
152  case "string":
153  propertyValue = value;
154  break;
155  case "System.Int32":
156  case "int":
157  {
158  var conversionSuccessful = int.TryParse(value, out int tmp);
159  if (conversionSuccessful)
160  {
161  propertyValue = tmp;
162  }
163 
164  break;
165  }
166  case "System.Double":
167  case "double":
168  {
169  var conversionSuccessful =
170  double.TryParse(value, NumberStyles.Any, CultureInfo.InvariantCulture, out var tmp);
171  if (conversionSuccessful)
172  {
173  propertyValue = tmp;
174  }
175 
176  break;
177  }
178  default: throw new InvalidCastException("Requested type not supported!");
179  }
180 
181  return propertyValue;
182  }

189  public void SaveSettings(SettingsViewModel settings)
190  {
191  var propertyInfos = settings.GetType().GetProperties();
192  foreach (var propertyInfo in propertyInfos)
193  {
194  // We don't want to save some settings every time, because they depend on something other than the
195  // settings window.
196  // For example, we don't want to update the settings related to the main window's state, because
197  // those are updated explicitly by a dedicated method
198  var isExplicitSaveOnlyAttribute =
199  propertyInfo.PropertyType.GetCustomAttribute<ExplicitSaveOnlyAttribute>();
200  if (isExplicitSaveOnlyAttribute is {IsExplicitSaveOnly: true})
201  {
202  continue;
203  }
204 
205  var propertiesToSave = propertyInfo.PropertyType.GetProperties();
206  foreach (var property in propertiesToSave)
207  {
208  var isNonPersistent = property.GetCustomAttribute<NonPersistentSettingAttribute>();
209  if (isNonPersistent is {IsNonPersistent: true})
210  {
211  continue;
212  }
213  string valueString = null;
214  var propertyName = property.Name;
215  var propertyValue = property.GetValue(propertyInfo.GetValue(settings));
216 
217  if (propertyValue != null)
218  {
219  valueString = propertyValue.ToString();
220  }
221 
222  _repository.UpdateSetting(propertyName, valueString);
223  }
224  }
